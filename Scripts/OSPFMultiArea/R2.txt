! --- Basic Configuration ---
hostname R2
enable secret cisco
line vty 0 4
 password cisco
 login
exit

! --- Interface Configuration ---
interface Loopback0
 ip address 2.2.2.2 255.255.255.255
!
interface GigabitEthernet0/0
 ip address 10.1.2.2 255.255.255.0  ! Link to R1
 no shutdown
!
interface GigabitEthernet0/1
 ip address 10.2.3.2 255.255.255.0  ! Link to R3
 no shutdown
exit

! --- OSPF Configuration ---
router ospf 1
 router-id 2.2.2.2
 network 10.1.2.0 0.0.0.255 area 1  ! Area 1 Interface
 network 10.2.3.0 0.0.0.255 area 0  ! Area 0 Interface
 network 2.2.2.2 0.0.0.0 area 0
exit

! --- Verification ---
show ip ospf border-routers
show ip ospf database
